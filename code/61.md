题目：打印出杨辉三角形，10行。
```c
#include<stdio.h>

int main()
{
	int a[10][10];
	for (int i = 0; i < 10; ++i)
	{
		a[i][0] = 1;
		a[i][i] = 1;
	}
	for (int i = 2; i < 10; ++i)
	{
		for (int j = 1; j < i; ++j)
		{
			a[i][j] = a[i - 1][j - 1] + a[i - 1][j];
		}
	}
	for (int i = 0; i < 10; ++i)
	{
		for (int j = 0; j <= i; ++j)
		{
			printf("%d ", a[i][j]);
		}
		printf("\n");
	}

	return 0;
}
```
