题目：打印出如下图案（菱形）。
```c
   *
  ***
 *****
*******
 *****
  ***
   *
```

```c
#include<stdio.h>

int main()
{
	for (int i = 0; i < 4; ++i)
	{
		for (int space = 3 - i; space > 0; --space)
		{
			printf(" ");
		}
		for (int j = 0; j <= 2 * i; ++j)
		{
			printf("*");
		}
		printf("\n");
	}

	for (int i = 3; i > 0; --i)
	{
		for (int space = 4 - i; space > 0; --space)
		{
			printf(" ");
		}

		for (int j = 2 * (i - 1); j >= 0; --j)
		{
			printf("*");
		}
		printf("\n");
	}

	return 0;
}
```
