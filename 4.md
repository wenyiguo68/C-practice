			break;

		case 12:
			cnt = 31 + 29 + 31 + 30 + 31 + 30 + 31 + 31 + 30 + 31 + 30 + day;
			break;
		}
	}
	else
	{
		switch (month)
		{
		case 1:
			cnt = day;

			cnt = 31 + day;

			cnt = 31 + 28 + day;

			break;

			cnt = 31 + 28 + 31 + 30 + 31 + 30 + 31 + 31 + day;

			break;

		case 12:
			cnt = 31 + 28 + 31 + 30 + 31 + 30 + 31 + 31 + 30 + 31 + 30 + day;

	}

	printf("这一天是这一年的第%d天", cnt);
